#include <stdio.h>
void Duplicates(int arr[], int n) {
    int i = 0;
    while (i < n) {
        int count = 1;
        while ((i < n - 1) && (arr[i] == arr[i + 1])) {
            count++;
            i++;
        }
        if (count > 1) {
            printf("%d->%d", arr[i], count);
            if (i < n - 1) {
                printf(",");
            }
        }
        i++;
    }
}
int main() {
    int a = 12;
    int arr[] = {1, 1, 1, 2, 4, 4, 4, 4, 5, 6, 9, 9};
    printf("Sample Output: ");
    Duplicates(arr, a);
    return 0;
}
